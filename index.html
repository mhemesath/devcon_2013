function randomPath(length, j) {
  var path = "",
      x = 10,
      y = 0;
  dotsy[j] = dotsy[j] || [];
  for (var i = 0; i < length; i++) {
      dotsy[j][i] = Math.round(Math.random() * 200);
      if (i) {
          x += 20;
          y = 240 - dotsy[j][i];
          path += "," + [x, y];
      } else {
          path += "M" + [10, (y = 240 - dotsy[j][i])] + "R";
      }
  }
  return path;
}

Imperative

vs.

Declarative

Imperative Approach: 

var numbers = [1,2,3,4,5]
var total = 0

for(var i = 0; i < numbers.length; i++) {
  total += numbers[i]
}
console.log(total) //=> 15
Declarative Approach: 

var numbers = [1,2,3,4,5]

var total = numbers.reduce(function(sum, n) {
  return sum + n
});
console.log(total) //=> 15
Declarative programming is powerful in user interfaces, graphics and animations.

Data-Driven Documents

D3 for short
How to draw: 
for (var i=0; i < data.length; i++) {
    var circle = document.createElement('circle');
    circle.cx = area.x(data[i]);
    circle.cy = area.y(data[i]);
    circle.r = 5;
    svg.appendChild(circle);
}
What to draw: 
svg.selectAll('circle').data(data)
    .enter().append('circle')
    .attr('cx', area.x())
    .attr('cy', area.y())
    .attr('r', 5);
Drawing a line
Define a scale 

var y = d3.scale.linear()
    .range([height, 0])
    .domain([min, max]);
Define a layout 

var data = [{x: 0, y: 5}, {x: 1, y: 3}, {x: 2, y: 10}]
var area = d3.svg.area()
    .x(function(d) { return x(d.x); })
    .y0(height)
    .y1(function(d) { return y(d.y); });
Render! 

svg.append('path').datum(data).attr('d', area);
